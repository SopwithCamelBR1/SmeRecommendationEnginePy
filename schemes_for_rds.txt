CREATE TABLE IF NOT EXISTS `smeme`.`users` (
  `userID` INT(6) NOT NULL,
  `firstname` VARCHAR(45) NOT NULL,
  `surname` VARCHAR(45) NOT NULL,
  `userRole` VARCHAR(45) NOT NULL,
  `userLocation` VARCHAR(45) NOT NULL,
  `userDeskLocation` VARCHAR(45) NULL,
  `email` VARCHAR(80) NOT NULL,
  `phone` VARCHAR(45) NOT NULL,
  `profilePhotoLink` VARCHAR(150) NULL,
  `CustomGroup` VARCHAR(150) NULL,
  PRIMARY KEY (`userID`),
  UNIQUE INDEX `email_UNIQUE` (`email` ASC)
);


CREATE TABLE `smeme`.`rec_technology` (
  `user_email` VARCHAR(80) NOT NULL,
  `rec_technologyID` INT(6) NOT NULL,
  INDEX `fk_useremail_tech_recs_idx` (`user_email` ASC),
  INDEX `fk_techid_rec_techs_idx` (`rec_technologyID` ASC),
  CONSTRAINT `fk_useremail_rec_techs`
    FOREIGN KEY (`user_email`)
    REFERENCES `smeme`.`users` (`email`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION,
  CONSTRAINT `fk_techid_rec_techs`
    FOREIGN KEY (`rec_technologyID`)
    REFERENCES `smeme`.`technologies` (`technologyID`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION);
